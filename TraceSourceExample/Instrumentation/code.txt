using System;

using TracingCore;usingTracingCore.TraceToPyDtos;namespace TraceSourceExample
{
    class Student
    {
        public Student()
        {
            TraceApi.TraceMethodEntry(6, "Student", new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(8);
            TraceApi.TraceMethodExit(8, false);
        }

        static Student()
        {
            TraceApi.RegisterClassLoad("TraceSourceExample.Student");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            TraceApi.TraceMethodEntry(13, "Main", new VariableData("args", args, args.GetType()));
            TraceApi.TraceData(15);
            int c;
            TraceApi.TraceData(16, new VariableData("c", "undefined", typeof(int)));
            var s = new Student();
            TraceApi.TraceData(17, new VariableData("s", s, s.GetType()));
            Console.WriteLine("Hello World!");
            TraceApi.TraceData(17);
            TraceApi.TraceMethodExit(17, true);
        }
    }
}
