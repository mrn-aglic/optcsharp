using System;
using TracingCore;
using TracingCore.TraceToPyDtos;

namespace ClrMD
{
    class Pozicija
    {
        public int x;
        public int y;
        public Pozicija(int x, int y)
        {
            TraceApi.TraceMethodEntry(10, "Pozicija", new VariableData("x", x, x.GetType()), new VariableData("y", y, y.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(12);
            this.x = x;
            TraceApi.TraceData(13);
            this.y = y;
            TraceApi.TraceData(13);
            TraceApi.TraceMethodExit(13, true);
        }

        static Pozicija()
        {
            TraceApi.RegisterClassLoad("ClrMD.Pozicija");
        }
    }

    class Sprite
    {
        private int x;
        private int y;
        private string smjer;
        public Sprite(int posX, int posY)
        {
            TraceApi.TraceMethodEntry(22, "Sprite", new VariableData("posX", posX, posX.GetType()), new VariableData("posY", posY, posY.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(24);
            x = posX;
            TraceApi.TraceData(25, new VariableData("x", x, x.GetType()));
            y = posY;
            TraceApi.TraceData(25, new VariableData("y", y, y.GetType()));
            TraceApi.TraceMethodExit(25, true);
        }

        public void PomakNa(int x, int y)
        {
            TraceApi.TraceMethodEntry(28, "PomakNa", new VariableData("x", x, x.GetType()), new VariableData("y", y, y.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(30);
            this.x = x;
            TraceApi.TraceData(31);
            this.y = y;
            TraceApi.TraceData(31);
            TraceApi.TraceMethodExit(31, true);
        }

        public void PomakNa(int x, int y, string smjer)
        {
            TraceApi.TraceMethodEntry(34, "PomakNa", new VariableData("x", x, x.GetType()), new VariableData("y", y, y.GetType()), new VariableData("smjer", smjer, smjer.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(36);
            this.x = x;
            TraceApi.TraceData(37);
            this.y = y;
            TraceApi.TraceData(38);
            this.smjer = smjer;
            TraceApi.TraceData(38);
            TraceApi.TraceMethodExit(38, true);
        }

        public void PomakNa(Pozicija poz)
        {
            TraceApi.TraceMethodEntry(41, "PomakNa", new VariableData("poz", poz, poz.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(43);
            this.x = poz.x;
            TraceApi.TraceData(44);
            this.y = poz.y;
            TraceApi.TraceData(44);
            TraceApi.TraceMethodExit(44, true);
        }

        public void PomakNa(params Pozicija[] pozicije)
        {
            TraceApi.TraceMethodEntry(47, "PomakNa", new VariableData("pozicije", pozicije, pozicije.GetType()), new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(49);
            foreach (var pozicija in pozicije)
            {
                TraceApi.TraceBlockEntry(50, new VariableData("pozicija", pozicija, pozicija.GetType()));
                TraceApi.TraceData(51);
                PomakNa(pozicija);
                TraceApi.TraceData(52);
                TraceApi.TraceBlockExit(52);
            }

            TraceApi.TraceMethodExit(49, true);
        }

        static Sprite()
        {
            TraceApi.RegisterClassLoad("ClrMD.Sprite");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            TraceApi.TraceMethodEntry(57, "Main", new VariableData("args", args, args.GetType()));
            TraceApi.TraceData(59);
            Sprite sprite = new Sprite(0, 0);
            TraceApi.TraceData(61, new VariableData("sprite", sprite, sprite.GetType()));
            Pozicija poz1 = new Pozicija(1, 1);
            TraceApi.TraceData(62, new VariableData("poz1", poz1, poz1.GetType()));
            Pozicija poz2 = new Pozicija(2, 3);
            TraceApi.TraceData(63, new VariableData("poz2", poz2, poz2.GetType()));
            Pozicija poz3 = new Pozicija(3, 4);
            TraceApi.TraceData(64, new VariableData("poz3", poz3, poz3.GetType()));
            sprite.PomakNa(poz1, poz2, poz3);
            TraceApi.TraceData(64);
            TraceApi.TraceMethodExit(64, true);
        }
    }
}
