using System;

namespace ClrMD
{
    class Pozicija
    {
        public int x;
        public int y;

        public Pozicija(int x, int y)
        {
            this.x = x;
            this.y = y;
        }
    }
    class Sprite
    {
        private int x;
        private int y;
        private string smjer;

        public Sprite(int posX, int posY)
        {
            x = posX;
            y = posY;
        }

        public void PomakNa(int x, int y)
        {
            this.x = x;
            this.y = y;
        }

        public void PomakNa(int x, int y, string smjer)
        {
            this.x = x;
            this.y = y;
            this.smjer = smjer;
        }

        public void PomakNa(Pozicija poz)
        {
            PomakNa(poz.x, poz.y);            
        }
    }
    class Program
    {
        static void Main(string[] args)
        {
            Sprite sprite = new Sprite(0, 0);
            
            sprite.PomakNa(0, 0);
            sprite.PomakNa(1, 1, "Gore");
            
            Pozicija poz = new Pozicija(2, 2);
            sprite.PomakNa(poz);
        }
    }
}
