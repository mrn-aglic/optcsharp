using System;

using TracingCore;usingTracingCore.TraceToPyDtos;public class Person
{
    public int age;
    public string name;
    static Person()
    {
        TraceApi.RegisterClassLoad("Person");
    }
}

class TestPerson
{
    static void Main()
    {
        TraceApi.TraceMethodEntry(10, "Main");
        TraceApi.TraceData(12);
        var person = new Person();
        TraceApi.TraceData(14, new VariableData("person", person, person.GetType()));
        Console.WriteLine("Name: {person.name}, Age: {person.age}");
        TraceApi.TraceData(16);
        // Keep the console window open in debug mode.
        Console.WriteLine("Press any key to exit.");
        TraceApi.TraceData(17);
        Console.ReadLine();
        TraceApi.TraceData(17);
        TraceApi.TraceMethodExit(17, true);
    }
}
