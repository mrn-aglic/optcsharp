using System;
using TracingCore;
using TracingCore.TraceToPyDtos;

public class Program
{
    public static void Main()
    {
        TraceApi.TraceMethodEntry(5, "Main");
        TraceApi.TraceData(7);
        Console.WriteLine("Hello World");
        TraceApi.TraceData(7);
        TraceApi.TraceMethodExit(7, true);
    }
}
