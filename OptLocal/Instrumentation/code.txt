using System;

using TracingCore;usingTracingCore.TraceToPyDtos;namespace TraceSourceExample
{
    class Student
    {
        public Student()
        {
            TraceApi.TraceBlockEntry(6, "Student", new VariableData("this", this, this.GetType()));
            TraceApi.TraceData(8);
            TraceApi.TraceBlockExit(8);
        }

        static Student()
        {
            TraceApi.RegisterClassLoad("TraceSourceExample.Student");
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            TraceApi.TraceMethodEntry(12, "Main", new VariableData("args", args, args.GetType()));
            TraceApi.TraceData(14);
            int c;
            TraceApi.TraceData(15, new VariableData("c", "undefined", typeof(int)));
            var s = new Student();
            TraceApi.TraceData(16, new VariableData("s", s, s.GetType()));
            Console.WriteLine("Hello World!");
            TraceApi.TraceData(16);
            TraceApi.TraceMethodExit(16, true);
        }
    }
}
